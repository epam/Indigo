CREATE TRIGGER {0}
	ON {1}
	FOR DELETE
AS
	declare @ID int;
	declare Cur cursor LOCAL for select {2} from deleted;

	open Cur

	fetch next from Cur into @ID

	while @@fetch_status = 0
	begin
		exec {3}._OnDeleteRecordTrigger '{1}', @ID

		fetch next from Cur into @ID
	end
  close Cur
  deallocate Cur
